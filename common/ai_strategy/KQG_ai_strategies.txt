attack_ningguang = {
	allowed = {
		original_tag = KQG
	}
	enable = {
		country_exists = NNG
		NNG = { has_war_with = KQG }
		OR = {
			strength_ratio = {
				tag = NNG
				ratio > 2
			}
			KQG = { enemies_strength_ratio < 0.85 }
		}
	}
	abort = {
		OR = {
			NOT = { country_exists = NNG }
			AND = {
				strength_ratio = {
					tag = NNG
					ratio < 1.5
				}
				KQG = { enemies_strength_ratio > 1 }
			}
		}
	}
	
	ai_strategy = {
		type = front_control
		tag = NNG
		ratio = 0.75
		priority = 100
		ordertype = front
		execution_type = rush_weak
		execute_order = yes 
		manual_attack = yes
	}
}
