on_actions = {
    on_state_control_changed = { 
        effect = {
            if = {
                limit = {
                    ROOT = QQR
                    FROM = MDQ
                }
                FROM.FROM = {
                    if = {
                        limit = {
                            has_state_flag = the_state_is_occupy_by_MDQ
                        }
                        modify_state_flag = {
                            flag = the_state_is_occupy_by_MDQ
                            value = 0
                        }
                        set_state_flag = the_state_is_occupy_by_QQR
                    }

                }
                
            }
        } 
    }
}
