#地块添加影响力 
INA_add_Impact_State_1 = 
{
	if = 
	{
		limit = {has_country_flag = Impact_give_large}
		add_to_variable = {
			global.KJG_impact_1 = 10
		}
		subtract_from_variable = {
			global.WSR_impact_1 = 10
		}
	}
	else = 
	{
		add_to_variable = {
			global.KJG_impact_1 = 5
		}
		subtract_from_variable = {
			global.WSR_impact_1 = 5
		}
	}
}
