#璃月内战第一阶段：中部军阀混战
kickstart_of_liyue_civil_war = { 
	KQG = {
		declare_war_on = {
			target = NNG
			type = annex_everything
		}
	}
}

unlock_liyue_advisors = {
	set_can_be_fired_in_advisor_role = {
		character = LYU_NingGuang 
		value = yes
	}
	set_can_be_fired_in_advisor_role = {
		character = LYU_Keqing 
		value = yes
	}
	set_can_be_fired_in_advisor_role = {
		character = LYU_Ganyu 
		value = yes
	}
}

reset_core_of_liyue = {
	ROOT = { add_research_slot = 1 }
	if = {
		limit = {
			has_template = "人民赤卫队"
		}
		delete_unit_template_and_units = "人民赤卫队"
		disband = yes
	}
	if = {
		limit = {
			has_template = "民兵"
		}
		delete_unit_template_and_units = "民兵"
		disband = yes
	}
	every_possible_country = {
		limit = { is_nominate_liyue = yes }
		every_state = {
			limit = { has_state_flag = is_Liyues_territory }
			remove_core_of = PREV
		}
	}
}
