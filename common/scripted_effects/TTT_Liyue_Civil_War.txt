liyue_adm_reconstruction_progress_upgrade = {
	if = {
		limit = {
			has_idea = Liyue_low_efficiency_administration_5
		}
		remove_ideas = Liyue_low_efficiency_administration_5
	}
	else_if = {
		limit = {
			has_idea = Liyue_low_efficiency_administration_4
		}
		swap_ideas = {
			add_idea = Liyue_low_efficiency_administration_5
			remove_idea = Liyue_low_efficiency_administration_4
		}
	}
	else_if = {
		limit = {
			has_idea = Liyue_low_efficiency_administration_3
		}

		}

LH_troika_sort = {
	resize_array = {
		array = troika_cha_array
		size = 3
		value = 0
	}
	resize_array = {
		array = troika_prestige_array
		size = 3
		value = 0
	}
	for_loop_effect = {
		end = 3
		value = v
		find_highest_in_array = {
			array = troika_prestige
			value = troika_prestige_highest_cha
			index = troika_cha
		}
		set_variable = {
			var = troika_cha_array^v
			value = troika_cha
		}
		set_variable = {
			var = troika_prestige_array^v
			value = troika_prestige_highest_cha
		}
		set_variable = {
			var = troika_prestige^troika_cha
			value = -1
		}
	}
}
