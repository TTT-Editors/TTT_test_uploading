calculation_BC_plan_pro = {
    for_loop_effect = {
        start = 0
        end = 4
        add_to_variable = {
            var = SUM_BC_pro_array^v
            value = SUM_BC_use_array^v
        }
    }
}
check_BC_plan_pro = {
    for_loop_effect = {
        start = 0
        end = 4
        if = {
            limit = {
                check_variable = {
                    var = SUM_BC_pro_array^v
                    value = 100
                    compare = greater_than_or_equals
                }
            }
            set_variable = {
                var = SUM_BC_comp_array^v
                value = 1
            }
        }
        if = {
            limit = {
                check_variable = {
                    var = SUM_BC_comp_array^v
                    value = 1
                    compare = equals
                }
            }
            set_variable = {
                var = SUM_BC_pro_array^v
                value = 100
            }
            set_variable = {
                var = SUM_BC_use_array^v
                value = 0
            }
        }
    }
}
