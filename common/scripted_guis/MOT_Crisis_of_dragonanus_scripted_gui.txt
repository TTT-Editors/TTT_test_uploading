scripted_gui = {

		}
        effects = {

            Crisis_of_dragonanus_decision2_click = {

                }

                }
            }
        }

        triggers = {
            Crisis_of_dragonanus_decision1_click_enabled = {
                not = {
                    has_country_flag = Lock_Crisis_of_dragonanus_decision_2
                }

                    has_country_flag = Lock_Crisis_of_dragonanus_decision_3
                }
            }

                not = {
                    has_country_flag = Lock_Crisis_of_dragonanus_decision_1

                }
            }

                not = {
                    has_country_flag = Lock_Crisis_of_dragonanus_decision_1

                }
            }
            Crisis_of_dragonanus_sign1_visible = {
