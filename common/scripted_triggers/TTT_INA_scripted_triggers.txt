	custom_trigger_tooltip = {
		tooltip = Narukami_Island_has_be_damaged

		    has_state_flag = Narukami_Island_damage2

Narukami_Island_consolidate_trigger = {
	OR = {
		AND = { tag = NAR  has_country_flag = open_Grand_Narukami_Shrine } 
		AND = { tag = KMR  has_country_flag = open_Yashiro_Commission }
		AND = { tag = KUJ  has_country_flag = open_Kujou }
	}
}

ISA_scripted_triggers = {
      if = {
			limit = {
				original_tag = INA
			}
			NOT = { has_country_flag = INA_NO_Same_type }
		}
		else_if = {
			limit = {
				original_tag = SNA
			}
			NOT = { has_country_flag = SNA_NO_Same_type }
		}
}

ISA_triggers_Declare_war = {
      if = {
			limit = {
				original_tag = INA
			}
			check_variable = { INA_junshizhichidu > 99 }
		    check_variable = { INA_minjianzhichidu > 99 }
		}
		else_if = {
			limit = {
				original_tag = SNA
			}
			check_variable = { SNA_junshizhichidu > 99 }
		    check_variable = { SNA_minjianzhichidu > 99 }
		}
}
