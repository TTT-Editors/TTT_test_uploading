country_event = {	##Liyue_Civil_War璃月内战
	id = super.1
	title = super.1.t
	desc = super.1.d
	picture = GFX_report_event_diplomatic_message

	is_triggered_only = yes
	
	fire_only_once = yes
	hidden = yes

	option = {
		name = super.1.a
		
		every_country = {
			limit = {
				OR = {
				    original_tag = LYU
				    original_tag = ANG
				    original_tag = RTG
				    original_tag = XRG
				    original_tag = KQG
				    original_tag = NNG
					original_tag = XOG
			    }
			}
			add_ideas = liyue_civilwar_for_Liyue
		}
        set_global_flag = Liyue_Civil_War_start
		set_global_flag = SupEvent_enable
		play_song = "SE_Liyue_civilwar"
		LYU = {
			set_cosmetic_tag = GANYULY
			TTT_open_book = yes
		}
		add_to_variable = {
			var = global.celestia_notice_level1 
			value = 5
		}
		celestia_notice_frame_effect = yes
	}
}
