		ai_chance = {
			base = 8
			
			#Badboy will never help you

				factor = 0

			}
			
			#Just on our border!
			modifier = {
				factor = 2
				OR = {
					tag = XRG
					tag = XOG
				}
			}
			
			#I have my war to deal with....
			modifier = {
				factor = 0.2
				has_war = yes
			}
			
			#too far away
			modifier = {
				factor = 0.5
				OR = {
					tag = LYU
					tag = ANG
					tag = RTG
				}
			}
			
			#isolation
			modifier = {
				factor = 0.1
				tag = ARF
			}
			
			#this is our duty
			modifier = {
				factor = 2
				is_major = yes
			}
		}
		add_to_war = {
			targeted_alliance = CUG
			enemy = ZDG
			hostility_reason = asked_to_join
		}
		CUG = { give_military_access = ROOT }

		name = CUG_event.27.a
	}
	
	#Decline
