country_event = { #死域随机事件
	id = sumeru.10000
	title = sumeru.10000.t
	desc = sumeru.10000.d

	trigger = {
		NOT = {
			has_country_flag = SUM_WitheringZone_is_clr
		}
	}
	mean_time_to_happen = {
		days = 55
		modifier = {
			factor = 1        
			tag = SUM
		}
	}
	hidden = yes
	immediate = {
		set_variable_to_random = {
			var = WitheringZone_event_index
			min = 1
			max = 17
			integer = yes
		}
	}

	option = {
		name = sumeru.10000.a
		add_to_variable = {
			var = WitheringZone_array_value^WitheringZone_event_index
			value = 10
			tooltip = random_WitheringZone_event_tt
		} 
		clamp_variable = {
			var = WitheringZone_array_value^WitheringZone_event_index
			min = 0
			max = 100
		}
		SUM_WitheringZone_flesh =yes
	}
}
