#如一阵风雪
country_event = {
    id = ZDG_events.8
	title = ZDG_events.8.t
    desc = ZDG_events.8.d
    fire_only_once=yes
	is_triggered_only = yes
	
	option = {
	    name = ZDG_events.8.a
		hidden_effect = {
			ZDG = { add_ideas = ZDG_likewhen_tathagatacame }
				ZDG = {
					transfer_units_fraction = {
						target = SNE
						size = 1
						keep_unit_leaders = {}
						keep_unit_leaders_trigger = {
							always = no
					}
			    }
				every_character = {
						set_nationality = SNE
			    }
				every_country = {
				    limit = {
					    is_subject_of = ZDG
					}
					ZDG = {
					    annex_country = {
						   target = PREV
						}
					}
				}
				every_neighbor_country = {
				    limit = {
					    is_nominate_liyue = yes
					}
					ZDG = {
					    declare_war_on = { target = PREV type = annex_everything }
					}
			    } 
		    }
		}
	}
}
