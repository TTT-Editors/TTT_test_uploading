#至冬获胜，璃月反至冬阵营成立
country_event = {
	id = liyue_civil_war.9
	title = liyue_civil_war.9.t
	desc = liyue_civil_war.9.d
	fire_only_once = yes
	hidden = yes
	
	trigger = {
		has_global_flag = Liyue_Civil_War
		tag = LYU
		ZDG = { exists = yes }
		LYU = { exists = yes }
		LYU = { NOT = { surrender_progress > 0.2 } }
		XOG = { exists = no }
		XRG = { exists = no }
	}
	
	mean_time_to_happen = { days = 45 }
	
	#最后一搏
	option = {
		name = liyue_civil_war.9.a
		set_global_flag = liyue_unified
		every_country = {
			limit = {
				is_nominate_liyue = yes
				NOT = { tag = ZDG }
			}
			if = {
				limit = {
					is_faction_leader = yes
					NOT = { tag = LYU }
				}
				dismantle_faction = yes
			}
			set_country_flag = ready_for_another_round
		}
		LYU = {
			create_faction = Liyue_united_faction
			declare_war_on = {
				target = ZDG
				type = annex_everything
			}
		}
		every_country = {
			limit = {
				has_country_flag = ready_for_another_round
				NOT = { has_government = hive_mind }
				is_subject = no
			}
			if = {
				limit = {
					has_war_with = ANG
				}
				white_peace = ANG
			}
			if = {
				limit = {
					has_war_with = RTG
					RTG = { NOT = { has_government = hive_mind } }
				}
				white_peace = RTG
			}
			if = {
				limit = {
					has_war_with = LYU
				}
				white_peace = LYU
			}
			if = {
				limit = {
					has_war_with = KQG
				}
				white_peace = KQG
			}
			if = {
				limit = {
					has_war_with = NNG
				}
				white_peace = NNG
			}
			if = {
				limit = {
					has_war_with = XQG
				}
				white_peace = XQG
			}
			LYU = { add_to_faction = PREV }
			add_to_war = {
				targeted_alliance = LYU
				enemy = ZDG
				hostility_reason = asked_to_join
			}
			set_country_flag = ready_for_another_round
		}
		news_event = { id = global_news.12 hours = 12 }
	}
}

				}

				tag = NNG
				XQG = { exists = yes }
			}

				add_to_faction = XQG

	}
}
