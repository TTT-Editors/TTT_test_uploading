#至冬获胜，璃月反至冬阵营成立
country_event = {
	id = liyue_civil_war.9
	title = liyue_civil_war.9.t
	desc = liyue_civil_war.9.d
	fire_only_once = yes
	hidden = yes
	
	is_triggered_only = yes
	
	#最后一搏
	option = {
		name = liyue_civil_war.9.a
		every_country = {
			limit = {
				has_country_flag = ready_to_join_pact
			}
			if = {
				limit = {
					is_faction_leader = yes
					NOT = { tag = LYU }
				}
				every_country = {
					if = {
						limit = { is_in_faction_with = PREV }
					}
					set_country_flag = ready_to_join_pact
					set_country_flag = ready_for_another_round
				}
				dismantle_faction = yes
			}
			set_country_flag = ready_for_another_round
		}
		every_country = {
			limit = {
				has_country_flag = ready_for_another_round
				NOT = { has_government = hive_mind }
				is_subject = no
			}
			if = {
				limit = { surrender_progress > 0.3 }
				every_owned_state = {
					limit = {
						NOT = { is_controlled_by = THIS.owner }
					}
					transfer_state_to = THIS.controller
				}
			}
			if = {
				limit = {
					has_war_with = ANG
				}
				white_peace = ANG
			}
			if = {
				limit = {
					has_war_with = RTG
					RTG = { NOT = { has_government = hive_mind } }
				}
				white_peace = RTG
			}
			if = {
				limit = {
					has_war_with = LYU
				}
				white_peace = LYU
			}
			if = {
				limit = {
					has_war_with = KQG
				}
				white_peace = KQG
			}
			if = {
				limit = {
					has_war_with = NNG
				}
				white_peace = NNG
			}
		}
		ROOT = {
			declare_war_on = {
				target = ZDG
				type = annex_everything
			}
		}
		every_country = {
			limit = { has_country_flag = ready_to_join_pact }
			ROOT = { add_to_faction = PREV }
			add_to_war = {
				targeted_alliance = ROOT
				enemy = ZDG
				hostility_reason = asked_to_join
			}
			clr_country_flag = ready_to_join_pact
			clr_country_flag = former_faction_members
		}
		news_event = { id = global_news.12 hours = 12 }
		news_event = { id = global_news.35 hours = 1 }
	}
}

#璃月反至冬阵营胜利，璃月进入决赛
country_event = {
	id = liyue_civil_war.10
	title = liyue_civil_war.10.t
	desc = liyue_civil_war.10.d
	fire_only_once = yes
	hidden = yes
	
	trigger = {
		has_global_flag = liyue_unified
		is_nominate_liyue = yes
		has_global_flag = zdg_has_come
		has_global_flag = Liyue_Civil_War
		ZDG = { exists = no }
	}
	
	is_triggered_only = yes
	
	#最后一战
	option = {
		name = liyue_civil_war.10.a
		every_country = {
			limit = {
				has_country_flag = ready_for_another_round
			}
			every_other_country = {
				limit = {
					has_country_flag = ready_for_another_round
					NOT = { has_war_with = PREV }
				}
				declare_war_on = {
					target = PREV
					type = annex_everything
				}
			}
		}
		news_event = { id = global_news.13 hours = 12 }
	}
}
